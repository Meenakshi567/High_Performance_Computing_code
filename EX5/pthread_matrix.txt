#include <stdio.h>
#include <pthread.h>

#define MAX 10

int A[MAX][MAX], B[MAX][MAX];
int sum[MAX][MAX], product[MAX][MAX];
int r1, c1, r2, c2;

// Thread function for addition
void* matrixAddition(void* arg)
{
    int i, j;

    if (r1 != r2 || c1 != c2)
    {
        printf("\nMatrix Addition not possible\n");
        pthread_exit(NULL);
    }

    for (i = 0; i < r1; i++)
    {
        for (j = 0; j < c1; j++)
        {
            sum[i][j] = A[i][j] + B[i][j];
        }
    }

    printf("\nMatrix Addition Result:\n");
    for (i = 0; i < r1; i++)
    {
        for (j = 0; j < c1; j++)
        {
            printf("%d ", sum[i][j]);
        }
        printf("\n");
    }

    pthread_exit(NULL);
}

// Thread function for multiplication
void* matrixMultiplication(void* arg)
{
    int i, j, k;

    if (c1 != r2)
    {
        printf("\nMatrix Multiplication not possible\n");
        pthread_exit(NULL);
    }

    for (i = 0; i < r1; i++)
    {
        for (j = 0; j < c2; j++)
        {
            product[i][j] = 0;
            for (k = 0; k < c1; k++)
            {
                product[i][j] += A[i][k] * B[k][j];
            }
        }
    }

    printf("\nMatrix Multiplication Result:\n");
    for (i = 0; i < r1; i++)
    {
        for (j = 0; j < c2; j++)
        {
            printf("%d ", product[i][j]);
        }
        printf("\n");
    }

    pthread_exit(NULL);
}

int main()
{
    pthread_t thread1, thread2;
    int i, j;

    printf("Enter rows and columns of Matrix A: ");
    scanf("%d %d", &r1, &c1);

    printf("Enter elements of Matrix A:\n");
    for (i = 0; i < r1; i++)
        for (j = 0; j < c1; j++)
            scanf("%d", &A[i][j]);

    printf("Enter rows and columns of Matrix B: ");
    scanf("%d %d", &r2, &c2);

    printf("Enter elements of Matrix B:\n");
    for (i = 0; i < r2; i++)
        for (j = 0; j < c2; j++)
            scanf("%d", &B[i][j]);

    // Create threads
    pthread_create(&thread1, NULL, matrixAddition, NULL);
    pthread_create(&thread2, NULL, matrixMultiplication, NULL);

    // Wait for threads to finish
    pthread_join(thread1, NULL);
    pthread_join(thread2, NULL);

    return 0;
}

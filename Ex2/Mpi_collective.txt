#include <stdio.h>
#include <stdlib.h>
#include <mpi.h>
#include <time.h>

#define N 20   // Total number of elements

int main(int argc, char *argv[]) {
    int rank, size;
    int number;          // Number to find frequency
    int *array = NULL;   // Full array (master only)
    int *subarray;       // Local array
    int local_count = 0;
    int total_count = 0;
    int chunk;

    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    chunk = N / size;
    subarray = (int *)malloc(chunk * sizeof(int));

    /* -------- Master Process -------- */
    if (rank == 0) {
        array = (int *)malloc(N * sizeof(int));

        printf("Enter the number to find frequency: ");
        scanf("%d", &number);

        // Generate random numbers
        srand(time(NULL));
        printf("Array elements:\n");
        for (int i = 0; i < N; i++) {
            array[i] = rand() % 10;
            printf("%d ", array[i]);
        }
        printf("\n");
    }

    /* Broadcast the number to all processes */
    MPI_Bcast(&number, 1, MPI_INT, 0, MPI_COMM_WORLD);

    /* Scatter array parts to all processes */
    MPI_Scatter(array, chunk, MPI_INT,
                subarray, chunk, MPI_INT,
                0, MPI_COMM_WORLD);

    /* Each process counts frequency in its subarray */
    for (int i = 0; i < chunk; i++) {
        if (subarray[i] == number)
            local_count++;
    }

    /* Reduce all local counts to master */
    MPI_Reduce(&local_count, &total_count, 1,
               MPI_INT, MPI_SUM, 0, MPI_COMM_WORLD);

    /* Master prints the result */
    if (rank == 0) {
        printf("\nTotal frequency of %d = %d\n", number, total_count);
    }

    MPI_Finalize();
    return 0;
}

#include <stdio.h>
#include <string.h>
#include <ctype.h>
#include <mpi.h>

#define MAX 100
#define TYPE1 1
#define TYPE2 2

// Function to check palindrome
int isPalindrome(char str[]) {
    int i, len = strlen(str);
    for (i = 0; i < len / 2; i++) {
        if (str[i] != str[len - i - 1])
            return 0;
    }
    return 1;
}

int main(int argc, char *argv[]) {
    int rank, size;
    char message[MAX];
    MPI_Status status;

    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    /* ---------------- Worker Processes ---------------- */
    if (rank != 0) {

        if (rank % 2 == 1) {
            // Odd ranked process → Type 1
            sprintf(message, "hello from rank %d", rank);
            MPI_Send(message, strlen(message) + 1, MPI_CHAR, 0, TYPE1, MPI_COMM_WORLD);
        } else {
            // Even ranked process → Type 2
            sprintf(message, "madam");   // palindrome example
            MPI_Send(message, strlen(message) + 1, MPI_CHAR, 0, TYPE2, MPI_COMM_WORLD);
        }

    }
    /* ---------------- Master Process ---------------- */
    else {
        for (int i = 1; i < size; i++) {
            MPI_Recv(message, MAX, MPI_CHAR, MPI_ANY_SOURCE, MPI_ANY_TAG,
                     MPI_COMM_WORLD, &status);

            int sender = status.MPI_SOURCE;
            int tag = status.MPI_TAG;

            if (tag == TYPE1) {
                // Convert to uppercase
                for (int j = 0; message[j]; j++)
                    message[j] = toupper(message[j]);

                printf("From Rank %d | Type 1 | Uppercase String: %s\n",
                       sender, message);
            }
            else if (tag == TYPE2) {
                // Check palindrome
                if (isPalindrome(message))
                    printf("From Rank %d | Type 2 | \"%s\" is a Palindrome\n",
                           sender, message);
                else
                    printf("From Rank %d | Type 2 | \"%s\" is NOT a Palindrome\n",
                           sender, message);
            }
        }
    }

    MPI_Finalize();
    return 0;
}

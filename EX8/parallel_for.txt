#include <stdio.h>
#include <stdlib.h>
#include <omp.h>

#define N 1000000   // Size of arrays

int main() {
    int i;
    double *A, *B, *C;

    // Allocate memory
    A = (double*) malloc(N * sizeof(double));
    B = (double*) malloc(N * sizeof(double));
    C = (double*) malloc(N * sizeof(double));

    // Initialize arrays
    for(i = 0; i < N; i++) {
        A[i] = i * 1.0;
        B[i] = i * 2.0;
    }

    // Parallel addition
    #pragma omp parallel for
    for(i = 0; i < N; i++) {
        C[i] = A[i] + B[i];
    }

    // Print some sample output
    printf("Sample results:\n");
    for(i = 0; i < 10; i++) {
        printf("C[%d] = %f\n", i, C[i]);
    }

    // Free memory
    free(A);
    free(B);
    free(C);

    return 0;
}
